§01 Pipeline at a glance

Twelve in-order frontend/backend stages, with out-of-order issue and writeback in the middle. Stages are listed oldest (left) to youngest (right).

§03 Resources & execution ports

Eight execution ports: two ALUs, one branch unit, one mul/div pipe, two memory pipes, two FPUs, and one vector pipe. Distributed schedulers (per IQ) feed the ports.

# -----------------------------------------------------------------------------
# Top-level pipeline shape
# 12-stage, 3-wide decode, 4-wide rename, 8-wide OoO issue, 3-wide retire
# -----------------------------------------------------------------------------
top:
  issue: 8           # peak OoO issue width across 8 execution ports
  commit: 3          # in-order retire width (matches 3-wide decode)
  inorder: 0         # out-of-order machine
  stages: [IF1, IF2, IF3, IB, ID, IR, DP, IS, RR, EX, WB, CM]

# -----------------------------------------------------------------------------
# Per-stage role declarations
# -----------------------------------------------------------------------------
stg:
  IF1: {role: fetch,     pc_gen: 1, bpred_lookup: 1}
  IF2: {role: fetch,     icache_access: 1}
  IF3: {role: fetch,     predecode: 1, ras_pop: 1}
  IB:  {role: fetch,     ibuffer: 1, align_16_32: 1}
  ID:  {role: decode,    decode_width: 3}
  IR:  {role: rename,    rename_width: 4, freelist_alloc: 1}
  DP:  {role: dispatch,  rob_alloc: 1, iq_dispatch: 1, lsq_alloc: 1}
  IS:  {role: issue,     wakeup: 1, select: 1, issue_width: 8}
  RR:  {role: issue,     rf_read: 1}
  EX:  {role: execute,   alu: 2, bru: 1, fpu: 2, lsu: 2, vu: 1}
  WB:  {role: writeback, rf_write: 1, cdb_broadcast: 1}
  CM:  {role: commit,    retire_width: 3, csr_update: 1, exception_point: 1}

# -----------------------------------------------------------------------------
# Resources — units, register files, queues, caches
# -----------------------------------------------------------------------------
res:

  prf_int: {type: regfile, rd_ports: 10, wr_ports: 5, entries: 96}
  prf_fp:  {type: regfile, rd_ports: 6,  wr_ports: 3, entries: 64}
  prf_vec: {type: regfile, rd_ports: 4,  wr_ports: 2, entries: 64}

  # Execution units (8 ports)
  alu0: {type: alu,    stage: EX, count: 1, latency: 1}
  alu1: {type: alu,    stage: EX, count: 1, latency: 1}
  bru0: {type: branch, stage: EX, count: 1, latency: 1}
  mdu0: {type: muldiv, stage: EX, count: 1, mul_lat: 3, div_lat: var}
  lsu0: {type: lsu,    stage: EX, count: 1}     # load pipe
  lsu1: {type: lsu,    stage: EX, count: 1}     # store / second load
  fpu0: {type: fpu,    stage: EX, count: 1, fma_lat: 4}
  fpu1: {type: fpu,    stage: EX, count: 1, fma_lat: 4}
  vu0:  {type: vector, stage: EX, count: 1}     # vector pipeline (RVV 0.7.1)

  # Caches (private L1, cluster-shared L2)
  icache: {type: cache, level: 1, size: 64K, ways: 2, line: 64, replace: fifo,
           predecode_bits_per_16b: 4}
  dcache: {type: cache, level: 1, size: 64K, ways: 4, line: 64,
           write_policy: writeback}
  l2:     {type: cache, level: 2, shared: cluster, ways: 16, coherent: 1}

  # TLBs
  itlb: {type: tlb, entries: 32}
  dtlb: {type: tlb, entries: 32}
  l2tlb: {type: tlb, entries: 1024, shared: 1}

# -----------------------------------------------------------------------------
# Instruction classes — production / writeback timing model
# -----------------------------------------------------------------------------
cls:
  ALU:
    match: [ADD, SUB, AND, OR, XOR, SLT, SLTU, ADDI, ANDI, ORI, XORI,
            SLTI, SLLI, SRLI, SRAI, LUI, AUIPC, ADDW, SUBW]
    src:  [rs1, "rs2?"]
    dst:  [rd]
    unit: alu0|alu1
    prod: EX
    wb:   WB
    cons: EX

  BR:
    match: [BEQ, BNE, BLT, BGE, BLTU, BGEU, JAL, JALR]
    src:  [rs1, "rs2?"]
    dst:  ["rd?"]
    unit: bru0
    resolve: EX
    prod: EX
    wb:   WB

  MUL:
    match: [MUL, MULH, MULHU, MULHSU, MULW]
    src:  [rs1, rs2]
    dst:  [rd]
    unit: mdu0
    prod: EX+2
    wb:   WB
    lat:  3

  DIV:
    match: [DIV, DIVU, REM, REMU, DIVW, DIVUW, REMW, REMUW]
    src:  [rs1, rs2]
    dst:  [rd]
    unit: mdu0
    prod: EX+var
    wb:   WB
    lat:  variable
    nonpipe: 1

  LOAD:
    match: [LB, LH, LW, LD, LBU, LHU, LWU, FLW, FLD]
    src:  [rs1]
    dst:  [rd]
    unit: lsu0|lsu1
    addr: EX
    mem:  EX+1
    prod: EX+2
    wb:   WB
    cons: EX

  STORE:
    match: [SB, SH, SW, SD, FSW, FSD]
    src:  [rs1, rs2]
    dst:  []
    unit: lsu0|lsu1
    addr: EX
    data: EX
    visible: CM   # store buffer drains after commit

  FPU:
    match: [FADD_S, FSUB_S, FMUL_S, FMADD_S, FADD_D, FSUB_D, FMUL_D, FMADD_D]
    src:  [rs1, rs2, "rs3?"]
    dst:  [rd]
    unit: fpu0|fpu1
    prod: EX+3
    wb:   WB
    lat:  4

  VEC:
    match: [VADD_VV, VMUL_VV, VLE, VSE, VFADD_VV]
    src:  ["vs1?", "vs2?", "rs1?"]
    dst:  ["vd?"]
    unit: vu0
    prod: EX+var
    wb:   WB

  CSR:
    match: [CSRRW, CSRRS, CSRRC, CSRRWI, CSRRSI, CSRRCI]
    src:  ["rs1?", csr]
    dst:  ["rd?", csr]
    unit: alu0
    prod: EX
    wb:   WB
    serialize: 1   # CSR ops serialize the pipe

  FENCE:
    match: [FENCE, FENCE_I, SFENCE_VMA]
    src:  []
    dst:  []
    serialize: 1
    drain: store_buffer

# -----------------------------------------------------------------------------
# Shared timing anchors
# -----------------------------------------------------------------------------
mov:

  ex_use:    EX
  mem_req:   EX
  mem_rsp:   EX+1
  rf_write:  WB
  commit:    CM

# -----------------------------------------------------------------------------
# Hazards (renaming removes most WAR/WAW; only RAW and memory hazards modeled)
# -----------------------------------------------------------------------------
haz:
  raw:
    - {from: WB, to: RR, via: fwd}        # CDB → RF read
    - {from: EX, to: EX, via: fwd}        # ALU → ALU same-cycle bypass
    - {from: EX+1, to: EX, via: fwd}      # load → use (1-cycle bubble best case)
  load_use:
    detect: IS
    when:  "producer=LOAD && consumer_cons=EX"
    act:   stall1
  struct:
    - {unit: mdu0, busy: 1, act: stall}   # MUL/DIV not fully pipelined for DIV
    - {unit: vu0,  busy: var, act: stall}
  mem_order:
    war: renamed
    waw: renamed
    raw_mem: lsq_disambiguate
  store_to_load_forward:
    enable: 1
    granularity: byte
    partial_overlap: replay
