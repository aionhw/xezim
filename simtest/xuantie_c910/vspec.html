sequence IntegerALU {
  instr A1: ADDI r3, r1, 12'h00A;    # placeholder example
  instr A2: ADD  r4, r1, r2;

  rule ADDIResult:
    assert (A1.valid == true && A1.opcode == 7'b0010011 && A1.funct3 == 3'b000)
    -> always (A1.result == A1.src1_val + A1.imm_sext);

    assert (A2.valid == true && A2.opcode == 7'b0110011 && A2.funct3 == 3'b000
                              && A2.funct7 == 7'b0000000)
    -> always (A2.result == A2.src1_val + A2.src2_val);

    assert (A2.issued == true)

  # x0 is architecturally hardwired to zero.
  rule X0HardwiredZero:
    assert (true)
    -> always (C910.xrf[0] == 64'h0000000000000000);

§04 Register renaming and physical destination allocation

sequencesequence Rename { 
sequence Rename {
  instr R1: ADDI r3, r1, 12'h001;
  instr R2: ADDI r3, r3, 12'h001;   # writes same arch reg → must get new PR

  # After rename, two consecutive writers of the same architectural register
  # must receive distinct physical destinations.
  rule WAWRenamed:
    assert (R1.renamed == true && R2.renamed == true && R1.arch_dest == R2.arch_dest)
    -> always (R1.phys_dest != R2.phys_dest);

  # Source operand of younger instruction is mapped to *previous* writer's
  # physical destination, if there is one.
  rule SourceRenameLookup:
    assert (R2.renamed == true && R2.arch_src1 == R1.arch_dest
                               && R1.in_flight == true)
    -> always (R2.phys_src1 == R1.phys_dest);

  # Old physical register returns to the freelist only at commit of overwriter.
  rule FreelistReleaseAtCommit:
    assert (R2.committed == true && R2.arch_dest == R1.arch_dest
                                 && R1.committed == true)
    -> eventually [0:1] (R2.old_phys_dest_freed == true);
}

§05 ROB allocation, retirement, and precise exceptions

3 RULES · 1 BLOCK
sequencesequence ROB { 
sequence ROB {
  instr Ix: GENERIC;

  rule ROBAllocAtDispatch:
    assert (Ix.dispatched == true)
    -> always (Ix.rob_index != ROB_INVALID);

  # ROB retires in program order.
  rule InOrderRetire:
    assert (Ix.committed == true && Iy.committed == true && Ix.older_than(Iy))
    -> always (Ix.commit_cycle <= Iy.commit_cycle);

  # Precise exception: when an exception is signaled at commit, all younger
  # in-flight instructions are flushed and architectural state is restored.
  rule PreciseException:
    assert (Ix.exception_at_commit == true)
    -> next (C910.flush == true
          && (forall Iy : Iy.older_than(Ix) == false && Iy.in_flight == true
                          -> Iy.killed == true));
}

§06 Out-of-order issue and wakeup

sequencesequence IssueWakeup { 
sequence IssueWakeup {
  instr P : GENERIC;          # producer
  instr C : GENERIC;          # consumer of P

  # A consumer can only issue once every source operand is ready
  # (either already in PRF or available via wakeup tag broadcast on the CDB).
  rule WakeupSelectReadiness:
    assert (C.issued == true && C.src1_tag == P.phys_dest)
    -> always (prev(P.writeback_en) == true
            || prev(P.cdb_broadcast) == true);

  # Age-based selection: among ready ops in the same IQ, the oldest issues
  # first (7-bit age vector).
  rule AgeOldestReady:
    assert (C.iq == P.iq && C.ready == true && P.ready == true
                         && P.age > C.age)
    -> always (P.selected_before_or_same_cycle_as(C));
}

sequencesequence StoreLoadForwarding { 
sequence StoreLoadForwarding {
  instr S1: SW [x1+0], x2;
  instr L1: LW x3, [x4+0];

  # Same-address forwarding from store buffer.
  rule StoreToLoadFwd:
    assert (S1.committed == false && S1.executed == true
                                  && S1.addr == L1.addr
                                  && S1.byte_mask == L1.byte_mask
                                  && S1.older_than(L1))
    -> always (L1.load_val == S1.data_val);

  # Different addresses: load reads memory.
  rule LoadFromMemory:
    assert (S1.addr != L1.addr)
    -> always (L1.load_val == { C910.dmem[L1.addr+7], C910.dmem[L1.addr+6],
                                C910.dmem[L1.addr+5], C910.dmem[L1.addr+4],
                                C910.dmem[L1.addr+3], C910.dmem[L1.addr+2],
                                C910.dmem[L1.addr+1], C910.dmem[L1.addr+0] });

  # Partial-overlap forwarding is not safe: triggers replay.
  rule PartialOverlapReplays:
    assert (S1.addr == L1.addr && S1.byte_mask != L1.byte_mask
                               && S1.older_than(L1)
                               && S1.committed == false)
    -> eventually [1:4] (L1.replay == true);
}

§08 Branch prediction, resolution, and recovery

sequencesequence BranchPredict { 
sequence BranchPredict {
  instr B1: BEQ r1, r2, +offset;

  # Prediction is consulted at IF1.
  rule PredictAtFetch:
    assert (B1.fetched == true)
    -> always (B1.prediction_made_at_stage == IF1);

  # Branch outcome is resolved at EX. On mispredict, front-end is redirected to IF1.
  rule MispredictRedirect:
    assert (B1.resolved == true && B1.predicted_taken != B1.actually_taken)
    -> eventually [1:1] (C910.flush == true
                       && B1.redirect_stage == IF1);

  # Predictor state is updated only at commit (not on speculative execution).
  rule PredictorUpdateAtCommit:
    assert (B1.committed == true)
    -> next (B1.bpred_state_updated == true);

  # Return-address stack: a call pushes link, a return pops the same address
  # in well-nested programs (modulo overflow).
  rule RASPushPop:
    assert (B1.is_call == true && B1.committed == true)
    -> next (C910.ras_top == B1.return_addr);
}
sequencesequence ReplayPredictor { 
sequence ReplayPredictor {
  instr LD: LW r1, [r10];

  # If a load class previously triggered a memory-order replay, the replay
  # predictor defers issuing further instances until older stores resolve.
  rule DeferOnPriorViolation:
    assert (LD.prior_violation == true && LD.dispatched == true)
    -> always (LD.defer_until_older_stores_resolved == true);
}

# V-Spec v1.0  —  Xuantie C910 (T-Head / Alibaba)
# Verifiable Specification: architectural and microarchitectural contract.
# Includes CPU rules, SoC/interconnect rules, and reset/power/retention rules.

# -----------------------------------------------------------------------------
# §1  Architectural and microarchitectural state
# -----------------------------------------------------------------------------
state C910 {
  # --- Architectural state ---
  reg pc       : bv[64];
  reg IR       : bv[32];                # current decoded instruction word
  reg xrf      : array [32] of bv[64];  # integer architectural reg file
  reg frf      : array [32] of bv[64];  # FP architectural reg file
  reg vrf      : array [32] of bv[128]; # vector architectural reg file (model)

  # --- Privileged state (subset) ---
  reg mstatus  : bv[64];
  reg mcause   : bv[64];
  reg mepc     : bv[64];
  reg mtvec    : bv[64];
  reg satp     : bv[64];                # Sv39 paging
  reg priv     : bv[2];                 # M/S/U

  # --- Memory ---
  reg dmem     : array [0xFFFFFFFFFF] of bv[8];   # 40-bit PA byte memory

  # --- Microarchitectural observables ---
  reg rob_head : bv[8];
  reg rob_tail : bv[8];
  reg ldq_occ  : bv[6];                 # 0..32
  reg stq_occ  : bv[6];                 # 0..24
  reg flush    : bool;

  # --- Reset / clock / power ---
  reg core_reset_n  : bool;
  reg core_clk_en   : bool;
  reg core_ready    : bool;
  reg pd_core_on    : bool;
  reg pd_core_retention : bool;
  reg pd_l2_on      : bool;
}

# §2  Decode and immediate handling (with explicit width adaptation)
# =============================================================================
sequence Decode {
  instr I1: GENERIC;

  # I-type immediate: bits [31:20], sign-extend to 64 (RISC-V convention).
  rule ITypeImmSlice:
    assert (I1.valid == true && I1.fmt == I_TYPE)
    -> always (I1.imm12 == C910.IR[31:20]);

  rule ITypeImmSext:
    assert (I1.valid == true && I1.fmt == I_TYPE)
    -> always (I1.imm_sext ==
               { 52'b0000000000000000000000000000000000000000000000000000,
                 I1.imm12 });   # placeholder for sign-extension helper

  # Opcode is the low 7 bits.
  rule OpcodeSlice:
    assert (I1.valid == true)
    -> always (I1.opcode == C910.IR[6:0]);

  rule RegFieldSlices:
    assert (I1.valid == true)
    -> always (I1.rd  == C910.IR[11:7]
            && I1.rs1 == C910.IR[19:15]
            && I1.rs2 == C910.IR[24:20]);
}


# =============================================================================
# §3  Integer ALU semantics (uses explicit width-adapted helpers)
# =============================================================================
sequence IntegerALU {
  instr A1: ADDI r3, r1, 12'h00A;    # placeholder example
  instr A2: ADD  r4, r1, r2;

  rule ADDIResult:
    assert (A1.valid == true && A1.opcode == 7'b0010011 && A1.funct3 == 3'b000)
    -> always (A1.result == A1.src1_val + A1.imm_sext);

  rule ADDResult:
    assert (A2.valid == true && A2.opcode == 7'b0110011 && A2.funct3 == 3'b000
                              && A2.funct7 == 7'b0000000)
    -> always (A2.result == A2.src1_val + A2.src2_val);

  # ALU is single-cycle: writeback within 1 cycle after issue.
  rule ALULatency:
    assert (A2.issued == true)
    -> eventually [1:1] (A2.writeback_en == true && A2.writeback_data == A2.result);

  # x0 is architecturally hardwired to zero.
  rule X0HardwiredZero:
    assert (true)
    -> always (C910.xrf[0] == 64'h0000000000000000);
}


# =============================================================================
# §4  Register renaming and physical destination allocation
# =============================================================================
sequence Rename {
  instr R1: ADDI r3, r1, 12'h001;
  instr R2: ADDI r3, r3, 12'h001;   # writes same arch reg → must get new PR

  # After rename, two consecutive writers of the same architectural register
  # must receive distinct physical destinations.
  rule WAWRenamed:
    assert (R1.renamed == true && R2.renamed == true && R1.arch_dest == R2.arch_dest)
    -> always (R1.phys_dest != R2.phys_dest);

  # Source operand of younger instruction is mapped to *previous* writer's
  # physical destination, if there is one.
  rule SourceRenameLookup:
    assert (R2.renamed == true && R2.arch_src1 == R1.arch_dest
                               && R1.in_flight == true)
    -> always (R2.phys_src1 == R1.phys_dest);

  # Old physical register returns to the freelist only at commit of overwriter.
  rule FreelistReleaseAtCommit:
    assert (R2.committed == true && R2.arch_dest == R1.arch_dest
                                 && R1.committed == true)
    -> eventually [0:1] (R2.old_phys_dest_freed == true);
}


# =============================================================================
# §5  ROB allocation, retirement, and precise exceptions
# =============================================================================
sequence ROB {
  instr Ix: GENERIC;

  rule ROBAllocAtDispatch:
    assert (Ix.dispatched == true)
    -> always (Ix.rob_index != ROB_INVALID);

  # ROB retires in program order.
  rule InOrderRetire:
    assert (Ix.committed == true && Iy.committed == true && Ix.older_than(Iy))
    -> always (Ix.commit_cycle <= Iy.commit_cycle);

  # Precise exception: when an exception is signaled at commit, all younger
  # in-flight instructions are flushed and architectural state is restored.
  rule PreciseException:
    assert (Ix.exception_at_commit == true)
    -> next (C910.flush == true
          && (forall Iy : Iy.older_than(Ix) == false && Iy.in_flight == true
                          -> Iy.killed == true));
}


# =============================================================================
# §6  Out-of-order issue and wakeup
# =============================================================================
sequence IssueWakeup {
  instr P : GENERIC;          # producer
  instr C : GENERIC;          # consumer of P

  # A consumer can only issue once every source operand is ready
  # (either already in PRF or available via wakeup tag broadcast on the CDB).
  rule WakeupSelectReadiness:
    assert (C.issued == true && C.src1_tag == P.phys_dest)
    -> always (prev(P.writeback_en) == true
            || prev(P.cdb_broadcast) == true);

  # Age-based selection: among ready ops in the same IQ, the oldest issues
  # first (7-bit age vector).
  rule AgeOldestReady:
    assert (C.iq == P.iq && C.ready == true && P.ready == true
                         && P.age > C.age)
    -> always (P.selected_before_or_same_cycle_as(C));
}

# §7  Load / Store / memory ordering
# =============================================================================
sequence StoreLoadForwarding {
  instr S1: SW [x1+0], x2;
  instr L1: LW x3, [x4+0];

  # Same-address forwarding from store buffer.
  rule StoreToLoadFwd:
    assert (S1.committed == false && S1.executed == true
                                  && S1.addr == L1.addr
                                  && S1.byte_mask == L1.byte_mask
                                  && S1.older_than(L1))
    -> always (L1.load_val == S1.data_val);

  # Different addresses: load reads memory.
  rule LoadFromMemory:
    assert (S1.addr != L1.addr)
    -> always (L1.load_val == { C910.dmem[L1.addr+7], C910.dmem[L1.addr+6],
                                C910.dmem[L1.addr+5], C910.dmem[L1.addr+4],
                                C910.dmem[L1.addr+3], C910.dmem[L1.addr+2],
                                C910.dmem[L1.addr+1], C910.dmem[L1.addr+0] });

  # Partial-overlap forwarding is not safe: triggers replay.
  rule PartialOverlapReplays:
    assert (S1.addr == L1.addr && S1.byte_mask != L1.byte_mask
                               && S1.older_than(L1)
                               && S1.committed == false)
    -> eventually [1:4] (L1.replay == true);
}

sequence MemoryOrdering {
  instr LD_A: LW r1, [r10];
  instr ST_B: SW [r11], r2;

  # Speculative load may pass an older unresolved store; on conflict, replay.
  rule SpeculativeLoadAllowed:
    assert (LD_A.older_than(ST_B) == false
         && LD_A.addr_resolved == true && ST_B.addr_resolved == false)
    -> always (LD_A.may_issue == true);

  rule MemViolationReplay:
    assert (ST_B.addr_resolved == true && LD_A.addr == ST_B.addr
                                       && ST_B.older_than(LD_A)
                                       && LD_A.executed_before(ST_B))
    -> eventually [1:6] (LD_A.replay_from_rob == true);
}

sequence FenceDrainsStores {
  instr F1: FENCE;
  rule FenceDrainsStoreBuffer:
    assert (F1.dispatched == true)
    -> eventually [1:64] (C910.stq_occ == 6'h00 && F1.completed == true);
}

# §8  Branch prediction, resolution, and recovery

sequence BranchPredict {
  instr B1: BEQ r1, r2, +offset;

  # Prediction is consulted at IF1.
  rule PredictAtFetch:
    assert (B1.fetched == true)
    -> always (B1.prediction_made_at_stage == IF1);

  # Branch outcome is resolved at EX. On mispredict, front-end is redirected to IF1.
  rule MispredictRedirect:
    assert (B1.resolved == true && B1.predicted_taken != B1.actually_taken)
    -> eventually [1:1] (C910.flush == true
                       && B1.redirect_stage == IF1);

  # Predictor state is updated only at commit (not on speculative execution).
  rule PredictorUpdateAtCommit:
    assert (B1.committed == true)
    -> next (B1.bpred_state_updated == true);

  # Return-address stack: a call pushes link, a return pops the same address
  # in well-nested programs (modulo overflow).
  rule RASPushPop:
    assert (B1.is_call == true && B1.committed == true)
    -> next (C910.ras_top == B1.return_addr);
}

sequence ReplayPredictor {
  instr LD: LW r1, [r10];

  # If a load class previously triggered a memory-order replay, the replay
  # predictor defers issuing further instances until older stores resolve.
  rule DeferOnPriorViolation:
    assert (LD.prior_violation == true && LD.dispatched == true)
    -> always (LD.defer_until_older_stores_resolved == true);
}
